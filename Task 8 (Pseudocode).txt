DECLARE input : String 
DECLARE firstChar : Char 

PRINT "Enter String: "
input <=== LCase(INPUT)

firstChar <=== Left(input, 1)

For count <=== 1 To Len(input)
    If firstChar = Mid(input, count, 1) Then
        Mid(input, count, 1) <=== "*"
    End If
Next

PRINT ""
PRINT firstChar , Mid(input, 2, Len(input))
